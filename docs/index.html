Risk Calculation Model

# Value at Risk (VaR) calculation
import numpy as np

def calculate_var(portfolio_value, confidence_level=0.05):
    returns = np.array([0.02, -0.01, 0.03, -0.02, 0.01])
    mean_return = np.mean(returns)
    std_return = np.std(returns)
    
    # Parametric VaR
    var_95 = portfolio_value * (mean_return - 1.645 * std_return)
    return abs(var_95)

portfolio = 1000000000  # $1B portfolio
daily_var = calculate_var(portfolio)
print(f"Daily VaR: ${daily_var:,.0f}")
